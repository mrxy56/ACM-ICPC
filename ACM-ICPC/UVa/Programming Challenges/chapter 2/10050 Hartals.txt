#include <cstdio>
#include <cstring>
using namespace std;
int main(void)
{
    int T,day,n,i,j,count;
    int h[110],d[3700];
    scanf("%d",&T);
    while(T--)
    {
        memset(h,0,sizeof(h));
        memset(d,0,sizeof(d));
        count=0;
        scanf("%d",&day);
        scanf("%d",&n);
        for(i=1;i<=n;i++)
        {
           scanf("%d",&h[i]);
           j=h[i];
           while(j<=day)
           {
               d[j]=1;
               j+=h[i];
           }
        }
        for(i=6;i<=day;i+=7)
           d[i]=0;
        for(i=7;i<=day;i+=7)
           d[i]=0;
        for(i=1;i<=day;i++)
           if(d[i])
              count++;
        printf("%d\n",count);
    }
    return 0;
}
