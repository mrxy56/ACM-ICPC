#include <stdio.h>
#include <string.h>
int main(void)
{
    char s[15]="";
    int i,j;
    int c[24][6]={{1,2,3,4,5,6},{1,4,2,5,3,6},{1,5,4,3,2,6},{1,3,5,2,4,6},
                          {2,6,3,4,1,5},{2,4,6,1,3,5},{2,1,4,3,6,5},{2,3,1,6,4,5},
                          {3,2,6,1,5,4,},{3,1,2,5,6,4},{3,5,1,6,2,4},{3,6,5,2,1,4},

                          {5,4,1,6,3,2},{5,6,4,3,1,2},{5,3,6,1,4,2},{5,1,3,4,6,2},
                          {6,2,4,3,5,1},{6,3,2,5,4,1},{6,5,3,4,2,1},{6,4,5,2,3,1}};
    while(fgets(s,15,stdin)!=NULL)
    {
        for(i=0;i<24;i++)
        {
            for(j=0;j<6;j++)
                if(s[c[0][j]-1]!=s[c[i][j]+5]) break;
            if(j==6) break;
        }
        if(i==24)
             printf("FALSE\n");
        else
             printf("TRUE\n");
    }
    return 0;
}
